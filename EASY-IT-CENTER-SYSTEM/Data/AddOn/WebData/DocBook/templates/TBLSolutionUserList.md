 [Timestamp]    DATETIME2                               NOT NULL  CONSTRAINT [DF_userList_timestamp] DEFAULT (getdate()),

 [Photo]        VARBINARY(max)                              NULL,

 CONSTRAINT   [IX_UserList]  UNIQUE      NONCLUSTERED ([UserName] asc) ,
