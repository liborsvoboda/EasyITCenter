 [UserId]         INT                                     NOT NULL,
 [Active]         BIT                                     NOT NULL,
 [Default]        BIT                                     NOT NULL,
 [TimeStamp]      DATETIME2                               NOT NULL  CONSTRAINT [DF_WebMenuList_TimeStamp] DEFAULT (getdate()),
 CONSTRAINT   [PK_WebMenuList]  PRIMARY KEY CLUSTERED    ([Id] asc) ,
