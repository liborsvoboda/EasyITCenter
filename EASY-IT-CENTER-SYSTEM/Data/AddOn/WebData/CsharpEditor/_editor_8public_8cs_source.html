319  /// </summary>
320  /// <param name="initialText">The initial text of the editor.</param>
321  /// <param name="preSource">The source code that should be prepended to the text of the document when compiling it.</param>
322  /// <param name="postSource">The source code that should be appended to the text of the document when compiling it.</param>
323  /// <param name="references">A list of <see cref="MetadataReference"/>s for which the compiled assembly will have bindings. Make sure to include an appropriate <see cref="DocumentationProvider"/>, if you would like documentation comments to appear in code completion windows. If this is <see langword="null"/>, references to all of the assemblies loaded in the current <see cref="AppDomain"/> will be added.</param>
324  /// <param name="compilationOptions">The compilation options used to compile the code. If this is <see langword="null"/>, a <c>new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary)</c> will be used.</param>
325  /// <param name="guid">A unique identifier for the document being edited. If this is <see langword="null"/>, a new <see cref="System.Guid"/> is generated. If the same identifier is used multiple times, the save history of the document will be available, even if the application has been closed between different sessions.</param>
326  /// <param name="additionalShortcuts">Additional application-specific shortcuts (for display purposes only - you need to implement your own logic).</param>
327  /// <returns>A fully initialised <see cref="Editor"/> instance.</returns>
328  public static async Task<Editor> Create(string initialText = "", string preSource = "", string postSource = "", IEnumerable<CachedMetadataReference> references = null, CSharpCompilationOptions compilationOptions = null, string guid = null, Shortcut[] additionalShortcuts = null)
329  {
330 
331  if (references == null)
332  {
333  List<CachedMetadataReference> referencesList = new List<CachedMetadataReference>();
334 
335  foreach (Assembly ass in AppDomain.CurrentDomain.GetAssemblies())
336  {
337  string location = null;
338 
339  try
340  {
341  location = ass.Location;
342  }
343  catch (NotSupportedException) { };
344 
345  if (!string.IsNullOrEmpty(location))
346  {
347  referencesList.Add(CachedMetadataReference.CreateFromFile(location, Path.Combine(Path.GetDirectoryName(location), Path.GetFileNameWithoutExtension(location) + ".xml")));
348  }
349  }
350 
351  references = referencesList;
352  }
353 
354  if (compilationOptions == null)
355  {
356  compilationOptions = new CSharpCompilationOptions(OutputKind.DynamicallyLinkedLibrary);
357  }
358 
359  if (string.IsNullOrEmpty(guid))
360  {
361  guid = System.Guid.NewGuid().ToString("N");
362  }
363  else
364  {
365  foreach (char c in Path.GetInvalidPathChars().Concat(Path.GetInvalidFileNameChars()))
366  {
367  if (guid.Contains(c))
368  {
369  throw new ArgumentException("The provided Guid \"" + guid + "\" is not valid!\nThe Guid must be a valid identifier for a path or a file.", nameof(guid));
370  }
371  }
372  }
373 
374  Editor tbr = new Editor(false);
375  await tbr.Initialize(initialText, preSource, postSource, references, compilationOptions, guid, additionalShortcuts ?? new Shortcut[0]);
376  return tbr;
377  }

557  /// <summary>
558  /// Compile the source code to an <see cref="Assembly"/>.
559  /// </summary>
560  /// <param name="synchronousBreak">The function to handle synchronous breakpoints. If this is <see langword="null" />, these breakpoints will be skipped. If you want to enable the default UI for breakpoints, use <see cref="SynchronousBreak(BreakpointInfo)"/> (or a function that calls it after performing additional operations).</param>
561  /// <param name="asynchronousBreak">The function to handle asynchronous breakpoints. If this is <see langword="null" />, these breakpoints will be skipped. If you want to enable the default UI for breakpoints, use <see cref="AsynchronousBreak(BreakpointInfo)"/> (or a function that calls it after performing additional operations).</param>
562  /// <returns>An <see cref="Assembly"/> containing the compiled code, or <see langword="null"/> if the compilation fails, as well as a <see cref="CSharpCompilation"/> that also contains information about any compilation errors.</returns>
563  public async Task<(Assembly Assembly, CSharpCompilation Compilation)> Compile(Func<BreakpointInfo, bool> synchronousBreak = null, Func<BreakpointInfo, Task<bool>> asynchronousBreak = null)
564  {
565  string source = null;
566 
567  await Dispatcher.UIThread.InvokeAsync(() =>
568  {
569  source = this.FullSource;
570  });
571 
572  SourceText sourceText = SourceText.From(source);
573 
574  ImmutableList<MetadataReference> references;
575 
576  lock (ReferencesLock)
577  {
578  references = References;
579  }
580 
581  SyntaxTree tree = await this.OriginalDocument.WithText(sourceText).GetSyntaxTreeAsync();
582 
583  CSharpCompilation comp = CSharpCompilation.Create("compilation", new[] { tree }, references, this.CompilationOptions);
584 
585  string debuggerGuid = "_" + System.Guid.NewGuid().ToString("N");
586 
587  if (synchronousBreak != null || asynchronousBreak != null)
588  {
589  List<(TextSpan, bool)> validBreakpoints = new List<(TextSpan, bool)>();
590 
591  foreach (int i in source.AllIndicesOf(Utils.BreakpointMarker))
592  {
593  SyntaxNode node = tree.GetRoot().FindNode(new TextSpan(i, 1));
594 
595  SyntaxNode fullNode = node;
596 
597  while (fullNode.Parent != null && !fullNode.Kind().IsStatement())
598  {
599  fullNode = fullNode.Parent;
600  }
601 
602  if (fullNode.Kind().IsStatement())
603  {
604  SyntaxNode methodNode = fullNode;
605 
606  while (methodNode.Parent != null && methodNode.Kind() != SyntaxKind.MethodDeclaration && methodNode.Kind() != SyntaxKind.ParenthesizedLambdaExpression && methodNode.Kind() != SyntaxKind.AnonymousMethodExpression)
607  {
608  methodNode = methodNode.Parent;
609  }
610 
611  if (methodNode.Kind() == SyntaxKind.MethodDeclaration || methodNode.Kind() == SyntaxKind.ParenthesizedLambdaExpression || methodNode.Kind() == SyntaxKind.AnonymousMethodExpression)
612  {
613  bool isAsync = false;
614 
615  if (methodNode.Kind() == SyntaxKind.MethodDeclaration)
616  {
617  MethodDeclarationSyntax method = (MethodDeclarationSyntax)methodNode;
618 
619  foreach (SyntaxToken token in method.Modifiers)
620  {
621  if (token.Text == "async")
622  {
623  isAsync = true;
624  }
625  }
626  }
627  else if (methodNode.Kind() == SyntaxKind.ParenthesizedLambdaExpression)
628  {
629  ParenthesizedLambdaExpressionSyntax method = (ParenthesizedLambdaExpressionSyntax)methodNode;
630 
631  if (method.AsyncKeyword.Text == "async")
632  {
633  isAsync = true;
634  }
635  }
636  else if (methodNode.Kind() == SyntaxKind.AnonymousMethodExpression)
637  {
638  AnonymousMethodExpressionSyntax method = (AnonymousMethodExpressionSyntax)methodNode;
639 
640  if (method.AsyncKeyword.Text == "async")
641  {
642  isAsync = true;
643  }
644  }
645 
646  if ((!isAsync && synchronousBreak != null) || (isAsync && asynchronousBreak != null))
647  {
648  validBreakpoints.Add((new TextSpan(i, Utils.BreakpointMarker.Length), isAsync));
649  }
650  }
651  }
652  }
653 
654  SyntaxTree debuggerTree = BreakpointInfo.GetDebuggerSyntaxTree(debuggerGuid);
655 
656  SemanticModel model = comp.GetSemanticModel(tree, false);
657 
658  for (int i = validBreakpoints.Count - 1; i >= 0; i--)
659  {
660  ILocalSymbol[] locals = (from el in model.LookupSymbols(validBreakpoints[i].Item1.Start) where el.Kind == SymbolKind.Local && !el.IsInaccessibleLocal(model, validBreakpoints[i].Item1.Start, tree.GetRoot().FindNode(validBreakpoints[i].Item1)) && !string.IsNullOrEmpty(el.Name) select (ILocalSymbol)el).ToArray();
661 
662  string breakpointSource = BreakpointInfo.GetBreakpointSource(validBreakpoints[i].Item1, locals, model, debuggerGuid, validBreakpoints[i].Item2);
663 
664  sourceText = sourceText.Replace(validBreakpoints[i].Item1, breakpointSource);
665  }
666 
667  string text = sourceText.ToString();
668 
669  tree = await this.OriginalDocument.WithText(sourceText).GetSyntaxTreeAsync();
670 
671  comp = CSharpCompilation.Create("compilation", new[] { debuggerTree, tree }, references, this.CompilationOptions);
672  }
673 
674  using MemoryStream ms = new MemoryStream();
675 
676  EmitResult result = comp.Emit(ms);
677 
678  if (!result.Success)
679  {
680  return (null, comp);
681  }
682  else
683  {
684  ms.Seek(0, SeekOrigin.Begin);
685  Assembly assembly = Assembly.Load(ms.ToArray());
686 
687  if (synchronousBreak != null || asynchronousBreak != null)
688  {
