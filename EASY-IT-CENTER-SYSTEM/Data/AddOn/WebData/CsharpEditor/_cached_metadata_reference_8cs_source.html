19 using Microsoft.CodeAnalysis;
20 using System.Collections.Generic;

22 namespace CSharpEditor

24  /// <summary>
25  /// This class represents a cached <see cref="MetadataReference"/>. When an instance of this class is created using the <see cref="CreateFromFile(string, string)"/> method,
26  /// we check whether a <see cref="CachedMetadataReference"/> to the same file has already been created and, in that case, return a reference to that object, rathern than creating a new one.
27  /// </summary>
28  public class CachedMetadataReference
29  {
30  private static Dictionary<string, CachedMetadataReference> CachedReferences = new Dictionary<string, CachedMetadataReference>();
31 
32  private MetadataReference Reference { get; }
33 
34  /// <summary>
35  /// Creates a new <see cref="CachedMetadataReference"/> wrapping the specified <paramref name="reference"/>.
36  /// </summary>
37  /// <param name="reference">The <see cref="MetadataReference"/> wrap in a new <see cref="CachedMetadataReference"/>.</param>
38  public CachedMetadataReference(MetadataReference reference)
39  {
40  this.Reference = reference;
41  }
42 
43  /// <summary>
44  /// Creates a new <see cref="CachedMetadataReference"/> from an assembly file (optionally including the XML documentation).
45  /// </summary>
46  /// <param name="path">The path to the assembly file.</param>
47  /// <param name="xmlDocumentationPath">The path to the XML documentation file for the assembly.</param>
48  /// <returns>
49  /// If a <see cref="CachedMetadataReference"/> has already been created from the same assembly file and the same XML documentation, a reference to the previously created object.
50  /// Otherwise, a new <see cref="CachedMetadataReference"/> wrapping a <see cref="MetadataReference"/> created from the specified assembly file.
51  /// </returns>
52  public static CachedMetadataReference CreateFromFile(string path, string xmlDocumentationPath = null)
53  {
54  string key = path + ":*:" + xmlDocumentationPath;
55 
56  if (CachedReferences.TryGetValue(key, out CachedMetadataReference cached))
57  {
58  return cached;
59  }
60  else
61  {
62  MetadataReference metadataReference = MetadataReference.CreateFromFile(path, documentation: XmlDocumentationProvider.CreateFromFile(xmlDocumentationPath));
63  CachedMetadataReference reference = new CachedMetadataReference(metadataReference);
64  CachedReferences.Add(key, reference);
65  return reference;
66  }
67  }
68 
69  /// <summary>
70  /// Converts a <see cref="CachedMetadataReference"/> into a <see cref="MetadataReference"/>.
71  /// </summary>
72  /// <param name="reference">The <see cref="CachedMetadataReference"/> to convert.</param>
73  public static implicit operator MetadataReference(CachedMetadataReference reference)
74  {
75  return reference.Reference;
76  }
77  }
78 }
