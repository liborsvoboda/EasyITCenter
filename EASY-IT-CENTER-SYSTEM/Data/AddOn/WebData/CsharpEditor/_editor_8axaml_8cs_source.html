366  internal async Task<BreakpointToggleResult> TryToggleBreakpoint(int lineStart, int lineEnd)
367  {
368  //string line = AvaloniaEditor.TextArea.Document.GetText(lineStart, lineEnd - lineStart);
369  string line = EditorControl.Text.ToString(new TextSpan(lineStart, lineEnd - lineStart));
370 
371  if (line.Contains(Utils.BreakpointMarker))
372  {
373  line = line.Replace(Utils.BreakpointMarker, "");
374  if (string.IsNullOrWhiteSpace(line))
375  {
376  await EditorControl.SetText(EditorControl.Text.WithChanges(new TextChange(new TextSpan(lineStart, lineEnd - lineStart + Environment.NewLine.Length), "")));
377  }
378  else
379  {
380  await EditorControl.SetText(EditorControl.Text.WithChanges(new TextChange(new TextSpan(lineStart, lineEnd - lineStart + Environment.NewLine.Length), line)));
381  }
382  return BreakpointToggleResult.Removed;
383  }
384  else
385  {
386  SyntaxTree tree = await EditorControl.GetSyntaxTree();
387 
388  TextSpan targetSpan = new TextSpan(lineStart, 1);
389 
390  SyntaxNode node = tree.GetRoot().FindNode(targetSpan);
391 
392  SyntaxNode fullNode = node;
393 
394  while (fullNode.Parent != null && !fullNode.Kind().IsStatement())
395  {
396  fullNode = fullNode.Parent;
397  }
398 
399  if (fullNode.Kind().IsStatement())
400  {
401  int charInd = tree.GetMappedLineSpan(fullNode.Span).StartLinePosition.Character;
402 
403  SyntaxTrivia commentTrivia = SyntaxFactory.Comment(Utils.BreakpointMarker + Environment.NewLine + new string(' ', charInd));
404 
405  SyntaxNode newNode = fullNode.WithLeadingTrivia(fullNode.GetLeadingTrivia().Add(commentTrivia));
406 
407  string newDocString = newNode.GetText().ToString();
408 
409  await EditorControl.SetText(EditorControl.Text.WithChanges(new TextChange(new TextSpan(fullNode.FullSpan.Start, fullNode.FullSpan.Length), newDocString)));
410 
411  //AvaloniaEditor.TextArea.Document.Replace(fullNode.FullSpan.Start, fullNode.FullSpan.Length, newDocString);
412  return BreakpointToggleResult.Added;
413  }
414  else
415  {
416  return BreakpointToggleResult.InvalidPosition;
417  }
418  }

421  internal async Task FormatText()
422  {
423  Document newDocument = EditorControl.Document;
424  SyntaxTree tree = await EditorControl.GetSyntaxTree();
425 
426  await EditorControl.SetText(Formatter.Format(tree.GetRoot(), newDocument.Project.Solution.Workspace).GetText());
427 
428  int caretOffset = EditorControl.CaretOffset;
429 
430  int initialCaretOffset = caretOffset;
431 
432  IEnumerable<TextChange> textChanges = await EditorControl.Document.GetTextChangesAsync(newDocument);
433 
434  foreach (TextChange change in from el in textChanges where el.Span.End <= initialCaretOffset select el)
435  {
436  caretOffset -= change.Span.Length - change.NewText.Length;
437  }
438 
439  EditorControl.SetSelection(caretOffset, 0);
440  }
