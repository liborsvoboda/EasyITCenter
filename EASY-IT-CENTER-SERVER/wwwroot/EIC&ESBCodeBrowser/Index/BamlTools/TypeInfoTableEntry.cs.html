using System;

namespace BamlTools
{
    public class TypeInfoTableEntry
    {
        int _typeId;
        String _fullName;

        public TypeInfoTableEntry(int typeId, string fullName)
        {
            _typeId = typeId;
            _fullName = fullName;
        }

        public int TypeId
        {
            get { return _typeId; }
        }

        public String FullName
        {
            get { return _fullName; }
        }

    }
}
