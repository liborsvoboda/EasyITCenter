ID:__demorpt-sales-by-countriesDESC:Here we select countries with total sales in it.SQL:SELECT Customers.Country AS Country, SUM(Orders.Freight) AS "Freight Sum" FROM (Orders AS Orders   INNER JOIN Customers AS Customers ON (Orders.CustomerID  = Customers.CustomerID ))WHERE(   '{{LASTYEAR}}-01-01' <= Orders.OrderDate AND Orders.OrderDate < '{{THISYEAR}}-01-01' )GROUP BY Customers.Country
